### 注：方括号包起来的是一个列表List，花括号包起来的是一个对象

```json
// 管理员
Administrator{
    token = "token"
    id = "id"
    passwd = "密码"
}

// 学生 student
Student{
    sid = "学生id"
	token = "token"
	name = "姓名"
	major = "专业"
	college = "学院"
	sex = "性别"
	passwd = "密码"
	phone = "手机号"
}

// 教师 teacher
Teacher{
    id = "教师id"
	name = "教师姓名"
	phone = "手机号"
	passwd = "密码"
}

// 教室 classroom
Classroom{
    cid = "教室id"
	token = "token"	
	address = "地址"
	courseSchedule = [{
    	Course
	}]
}

// 课程 course
Course{
    courseId = "课程号"
	name = "名字"
	teacher = "教师"
	site = "上课地点"
	date = ["上课时间 格式yyyy-MM-dd"]
	startSection = "开始节次，Integer型数据"
	endSection = "结束节次，Integer型数据"
	students = [{
        Student 		// 该课程的学生
    }]
}

// 二维码 qrcode
QRcode{
    code = "二维码的base64码"
    courseId = "对应的课程号"
    cid = "对应的教室id"
    uid = "二维码唯一标识符"
    time = "签到时间"
    count = "签到人数"
    sid = ["需要签到的学生id"]
}

// 二维码更新 qrcodeUpdate
QRCodeUpdate{
    uid = "二维码唯一标识符"
    students = [{
    	Student 		// 已经签到了的学生
	}]
}

// 签到结果查询 qrcodeResult
QRCodeResult{
    courseName = "对应的课程名称"
    courseId = "对应的课程号"
    date = "签到时间"
    teacher = "对应的教师"
    students = [{
    	Student 		// 已经签到的学生
	}]
}

// 课堂交流 chat
Chat{
    cid = "教室id"
    sid = "学生id"
    name = "学生姓名"
    time = "发送时间"
    content = "内容"
}

// 消息分发 message
Message{
    id = "消息id，不用填写，由数据库自动生成"
    title = "标题"
    content = "消息内容"
    time = "发布时间 格式yyyy-MM-dd HH:mm:ss"
    attachment = [{
    	Attachment 		// 附件，总文件大小不超过10MB
	}]
}

// 附件 attachment
Attachment{
    name = "名称"
    file = "multipart/form-data类型，单个文件大小不超过1MB"
}


// 课表
CourseSchedule{
	student = {Student}		// 对应学生的课表
	courses = [{
        Course		 // 课程列表
    }]
}

// 查询学生信息 queryStudent
QueryStudent{
    sid = "学生id"
    name = "学生姓名"
    major = "学生专业"
    college = "学生学院"
    phone = "学生手机号"
}

// 查询教室信息 queryClassroom
QueryClassroom{
    cid = "教室id"
    address = "教室地址"
}

// 查询课程信息
QueryCourse{
    id = "课程号"
    name = "课程名称"
    teacher = "任课教师"
}

// 用户请求类型
set{
    type = "student/teacher/administrator"
}

// 心跳包
HeatBeat{
    sid = "学生id"
    cid = "教室id"
}
```
